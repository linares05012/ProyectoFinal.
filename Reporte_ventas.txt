#!/bin/bash

ARCHIVO="ventas_compania_actualizado.csv"
REPORTE="Reporte_ventas.txt"

# Verificar si el archivo existe
function verificar_archivo {
    if [[ ! -f "$ARCHIVO" ]]; then
        echo " Archivo no encontrado: $ARCHIVO"
        exit 1
    fi
}

verificar_archivo

# FunciÃ³n: Total de ingresos por categorÃ­a
function ingresos_por_categoria {
    echo -e "\n Total de ingresos por categorÃ­a:" >> "$REPORTE"
    tail -n +2 "$ARCHIVO" | awk -F';' '{ ingresos[$4] += $7 } END { for (cat in ingresos) printf "â€¢ %s: $%.2f\n", cat, ingresos[cat] }' >> "$REPORTE"
}

# FunciÃ³n: Total de ingresos por mes
function ingresos_por_mes {
    echo -e "\n Total de ingresos por mes:" >> "$REPORTE"
    tail -n +2 "$ARCHIVO" | awk -F';' '{
        split($1, fecha, "-");
        mes = fecha[1] "-" fecha[2];
        ingresos[mes] += $7
    } END {
        for (m in ingresos) printf "â€¢ %s: $%.2f\n", m, ingresos[m]
    }' >> "$REPORTE"
}

# FunciÃ³n: Total de ingresos por cliente
function ingresos_por_cliente {
    echo -e "\n Total de ingresos por cliente:" >> "$REPORTE"
    tail -n +2 "$ARCHIVO" | awk -F';' '{ ingresos[$2] += $7 } END { for (c in ingresos) printf "â€¢ %s: $%.2f\n", c, ingresos[c] }' >> "$REPORTE"
}

# FunciÃ³n: Total de ingresos por departamento
function ingresos_por_departamento {
    echo -e "\n Total de ingresos por departamento:" >> "$REPORTE"
    tail -n +2 "$ARCHIVO" | awk -F';' '{ ingresos[$3] += $7 } END { for (d in ingresos) printf "â€¢ %s: $%.2f\n", d, ingresos[d] }' >> "$REPORTE"
}

# FunciÃ³n: Top 10 productos mÃ¡s vendidos
function top_10_productos {
    echo -e "\n Top 10 productos mÃ¡s vendidos (por cantidad):" >> "$REPORTE"
    tail -n +2 "$ARCHIVO" | awk -F';' '{
        gsub(/^ +| +$/, "", $5);
        productos[$5] += $6
    } END {
        for (p in productos) print productos[p], p
    }' | sort -rn | head -10 | awk '{ printf "â€¢ %s unidades - %s\n", $1, $2 }' >> "$REPORTE"
}

# FunciÃ³n: Generar reporte completo
function generar_reporte {
    echo "Generando reporte en $REPORTE..."
    echo "========== REPORTE DIARIO ==========" > "$REPORTE"
    ingresos_por_categoria
    ingresos_por_mes
    ingresos_por_cliente
    ingresos_por_departamento
    top_10_productos
    echo "Reporte generado exitosamente en $REPORTE."
}

# Mostrar menÃº
function mostrar_menu {
    echo -e "\n========== MENÃš DE REPORTES =========="
    echo "1) Generar reporte completo"
    echo "0) Salir"
    echo "======================================"
    echo -n "Selecciona una opciÃ³n: "
}

# Loop del menÃº
